## 案例
```go
package main

import "fmt"

func main() {
	s := test()
	fmt.Println("s:", s)
}

func hello() bool {
	fmt.Println("hello")
	return true
}

func test() bool {
	defer func() {
		fmt.Println("texting")
	}()
	return hello()
}

```
